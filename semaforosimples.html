// C++ code
        //
        int socrro = 0;
        
        void setup()
        {
          pinMode(12, OUTPUT);
          pinMode(4, OUTPUT);
          pinMode(8, OUTPUT);
        
          socrro = 0;
        }
        
        void loop()
        {
          while (!(socrro == 2)) {
            digitalWrite(12, HIGH);
            delay(9000); // Wait for 9000 millisecond(s)
            digitalWrite(4, HIGH);
            digitalWrite(12, LOW);
            delay(7000); // Wait for 7000 millisecond(s)
            digitalWrite(8, HIGH);
            digitalWrite(4, LOW);
            delay(2000); // Wait for 2000 millisecond(s)
            digitalWrite(8, LOW);
            socrro += 1;
          }
          digitalWrite(8, HIGH);
          delay(1000); // Wait for 1000 millisecond(s)
          digitalWrite(8, LOW);
          delay(1000); // Wait for 1000 millisecond(s)
        }
